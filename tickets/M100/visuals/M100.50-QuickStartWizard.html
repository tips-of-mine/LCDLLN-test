Template paramétré (wizard_template.json)

Extrait du template
{
  "version": 1,
  "id": "quick_start",
  "choices": {
    "climate": { ... 4 options },
    "relief":  { ... 4 options },
    "coast":   { ... 3 options },
    "poi":     { ... 4 options }
  },

  "operationTemplate": [
    {
      "type": "mountain_macro",
      "preset": "{{climate}}_{{relief}}",
      "polyline": "$auto_chain_for_{{relief}}",
      "heightMeters": "$base_height * {{relief.reliefMultiplier}}",
      "affectedBy": ["relief"]
    },
    {
      "type": "coastline",
      "preset": "{{coast}}",
      "if": "{{coast}} != 'inland'"
    },
    {
      "type": "place_cave",
      "catalogId": "cave_small_01",
      "if": "{{poi}} == 'cave'"
    },
    {
      "type": "place_dungeon",
      "dungeonId": "auto_generated_{{seed}}",
      "if": "{{poi}} == 'dungeon'"
    }
    // ... 4 autres opérations
  ]
}
Résolution → ZonePreset
WizardChoices choices = {
  .climate = "temperate",
  .relief  = "hills",
  .coast   = "moderate",
  .poi     = "cave",
  .seed    = 42
};

ZonePreset preset = resolver.Resolve(choices);

// preset.operations contient maintenant :
[
  { type: "mountain_macro",

    heightMeters: 300.0,  // baseHeight * 0.7
    affectedBy: ["relief"] },
  { type: "coastline",
    preset: "moderate",
    seaLevelMeters: 50 },
  { type: "river_network", ... },

]
